% PDF rendering tweaks for the cross-vendor heterogeneity guard manuscript.
% Two real problems being fixed:
%   1. Long monospace paths in inline code overflow the right margin.
%   2. Wide tables (Appendix A, Appendix C, the §6 pilot tables) overlap
%      cells because pandoc-generated longtable column widths don't fit.
% This header addresses both.

% --- Inline-code / URL overflow tweaks ---

\emergencystretch=3em
\hfuzz=2pt
\hbadness=2000
\vbadness=2000

\usepackage{xurl}
\Urlmuskip=0mu plus 1mu

% --- Wide-table handling ---

\usepackage{adjustbox}
\usepackage{tabularx}
\usepackage{longtable}
\usepackage{ragged2e}

% Tighten inter-column padding everywhere; gives wide tables more cell room.
\setlength{\tabcolsep}{3pt}
\renewcommand{\arraystretch}{1.05}

% No global table-font override. Wide tables (Table 1 in §6; Appendix A;
% Appendix C) get explicit `\begingroup\scriptsize ... \endgroup` wrappers
% in the markdown source. Narrow tables (§6 pilot data, etc.) stay at body
% font size where they remain readable.
%
% Earlier version of this file globally forced `\small` inside every
% longtable via \AtBeginEnvironment. That fired AFTER \begingroup\scriptsize
% from the per-table wraps and silently reverted them to \small, which is
% why Table 1 in §6 kept overflowing despite the wrap being applied.

% Page-break flexibility around section headers and tables.
\widowpenalty=1000
\clubpenalty=1000
\setlength{\LTpre}{0.5em}
\setlength{\LTpost}{0.5em}
